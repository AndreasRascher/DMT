    LOCAL PROCEDURE SetFieldValue@1000000025(VAR valueAsText@1000000000 : Text;fieldIndex@1000000001 : Integer);
    VAR
      currTableRef@1000000003 : RecordRef;
      currFldRef@1000000006 : FieldRef;
      _TableNo@1000000004 : Integer;
      field@1000000005 : Record 2000000041;
    BEGIN
      //* returns the field value as text from the current column (ExportFieldBuffer) of the global recordref (ExportRecRef)
      CLEAR(valueAsText);
      IF fieldIndex > MaxNoOfFields THEN
        EXIT;

      IF fieldIndex = 1 THEN BEGIN
        ExportFieldBuffer.FINDFIRST;
      END ELSE BEGIN
        ExportFieldBuffer.NEXT;
      END;

      CLEAR(valueAsText);
      // Format Field in XML Format, Zero numbers are empty (Save Space)
      CLEAR(currTableRef);
      CASE TRUE OF
        (ExportFieldBuffer.TableNo = DATABASE::Field) AND (ExportFieldBuffer."No." = 50000): BEGIN
          _TableNo := ExportRecRef.FIELD(1).VALUE;
          currTableRef.OPEN(_TableNo,FALSE);
          valueAsText := currTableRef.CAPTION;
        END;
        (ExportFieldBuffer.TableNo = DATABASE::Field) AND (ExportFieldBuffer."No." = 50001): BEGIN
          _TableNo := ExportRecRef.FIELD(1).VALUE;
          currTableRef.OPEN(_TableNo,FALSE);
          valueAsText := GetListOfKeyFieldIDs(currTableRef.NUMBER);
        END;
        //OptionString
        (ExportFieldBuffer.TableNo = DATABASE::Field) AND (ExportFieldBuffer."No." = 50002): BEGIN
          ExportRecRef.SETTABLE(field);
          IF field.Type = field.Type::Option THEN BEGIN
            currTableRef.OPEN(field.TableNo,FALSE);
            currFldRef := currTableRef.FIELD(field."No.");
            GetListOfOptionString(currFldRef,valueAsText);
          END;
        END;
        (ExportFieldBuffer.TableNo = DATABASE::Field) AND (ExportFieldBuffer."No."=50003): BEGIN
          ExportRecRef.SETTABLE(field);
          IF field.Type = field.Type::Option THEN BEGIN
            currTableRef.OPEN(field.TableNo,FALSE);
            currFldRef := currTableRef.FIELD(field."No.");
            //mask leading "
            IF (STRLEN(currFldRef.OPTIONCAPTION)>1) AND (COPYSTR(currFldRef.OPTIONCAPTION,1,2)=' ,') THEN BEGIN
              valueAsText := '" "'+COPYSTR(currFldRef.OPTIONCAPTION,2);
            END ELSE BEGIN
              valueAsText := currFldRef.OPTIONCAPTION;
            END;
          END;
        END;
        // No of Records
        (ExportFieldBuffer.TableNo = DATABASE::Field) AND (ExportFieldBuffer."No."=50004): BEGIN
          ExportRecRef.SETTABLE(field);
          valueAsText := GetNoOfRecordsInTable(field);
        END;
        ELSE valueAsText := GetFormattedFieldValue(ExportRecRef,ExportFieldBuffer."No.");
      END; // END_CASE

      valueAsText := ReplaceStr(valueAsText,FieldSeparator,'');  // Remove FieldSeparator from Value
      valueAsText := RemoveSpecialChars(valueAsText);

      // Add Closing Sign
      IF LastFieldNo <> ExportFieldBuffer."No." THEN
        valueAsText += FieldSeparator;
    END;

    LOCAL PROCEDURE SetTableFilterToExludeNonNormal@55001(VAR field@55001 : Record 2000000041) FilterExpr : Text;
    VAR
      AllObj@55004 : Record 2000000038;
      recRef@55002 : RecordRef;
      fieldRef@55003 : FieldRef;
      TableNos@55005 : TEMPORARY Record 2000000026;
      Object@1000000000 : Record 2000000001;
    BEGIN
      //* depending on the version of nav some tables must be excluded
      IF AllObj.GET(AllObj."Object Type"::Table,2000000136) THEN BEGIN

        recRef.OPEN(2000000136);

        IF recRef.FIELDEXIST(11) THEN BEGIN //TableMetadata.TableType
          recRef.RESET;
          fieldRef := recRef.FIELD(11);
          fieldRef.SETFILTER('>0');
          IF recRef.FINDSET THEN
            REPEAT
              fieldRef := recRef.FIELD(1);
              TableNos.Number := fieldRef.VALUE;
              IF TableNos.INSERT THEN;
            UNTIL recRef.NEXT = 0;
        END;

        IF recRef.FIELDEXIST(13) THEN BEGIN //TableMetadata.ObsoleteState
          recRef.RESET;
          fieldRef := recRef.FIELD(13);
          fieldRef.SETFILTER('>0');
          IF recRef.FINDSET THEN REPEAT
              fieldRef := recRef.FIELD(1);
              TableNos.Number := fieldRef.VALUE;
              IF TableNos.INSERT THEN;
          UNTIL recRef.NEXT = 0;
        END;

      END;

      //* Exclude not compiled tables
      Object.RESET;
      Object.SETRANGE(Type,Object.Type::Table);
      Object.SETRANGE("Company Name",'');
      Object.SETRANGE(Compiled,FALSE);
      IF Object.FINDSET THEN REPEAT
        TableNos.Number := Object.ID;
        IF TableNos.INSERT THEN;
      UNTIL Object.NEXT = 0;

      //* Build Filter
      IF TableNos.FINDSET THEN REPEAT
        FilterExpr += '&<>'+FORMAT(TableNos.Number);
      UNTIL TableNos.NEXT = 0;
      FilterExpr := DELCHR(FilterExpr,'<','&');

      IF FilterExpr <> '' THEN
        field.SETFILTER(TableNo,FilterExpr);
    END;
