    LOCAL PROCEDURE CheckTablesAndCreateActualTableFilter@55000();
    VAR
      ObjectNotInLicenseMsg@55002 : TextConst 'DEU=Achtung: Tabelle %1 - %2 ist nicht in der Lizenz.;ENU=Warning: Table %1 - %2 is not in license.';
      EmptyTableMsg@55001 : TextConst 'DEU=Achtung: Tabelle %1 - %2 ist leer. Fr leere Tabellen wird keine Exportdatei erstellt.';
      AllObjWithCaption@55003 : Record 2000000058;
      TableIsNotCompiledMsg@55004 : TextConst 'DEU=Achtung: Tabelle %1 ist nicht kompiliert;ENU=Warning: Table %1 is not compiled';
      isCompiled@55005 : Boolean;
      isEmpty@55006 : Boolean;
      isInLicense@55007 : Boolean;
    BEGIN
      //* check if tables in filter are actual available
      IF (ActualExportTableFilter <> '') THEN // is predefined in filter
        EXIT;
      IF (ExportTableIDFilter = '') THEN
        ERROR('ExportTableIDFilter Empty');


      AllObjWithCaption.SETRANGE("Object Type",AllObjWithCaption."Object Type"::TableData);
      AllObjWithCaption.SETFILTER("Object ID",ExportTableIDFilter);
      IF AllObjWithCaption.FINDSET THEN REPEAT
        // check properties
        CLEAR(isCompiled);
        CLEAR(isEmpty);
        CLEAR(isInLicense);
        isCompiled := IsTableCompiled(AllObjWithCaption."Object ID");
        IF isCompiled THEN BEGIN
          isEmpty := IsTableEmpty(AllObjWithCaption."Object ID");
          isInLicense := IsTableInLicense(AllObjWithCaption."Object ID");
        END;
        // property warnings
        IF NOT isCompiled THEN BEGIN
          MESSAGE(TableIsNotCompiledMsg,AllObjWithCaption."Object ID");
        END ELSE BEGIN
          IF NOT isInLicense THEN
            MESSAGE(ObjectNotInLicenseMsg,AllObjWithCaption."Object ID",AllObjWithCaption."Object Caption");
          IF isEmpty THEN
            MESSAGE(EmptyTableMsg,AllObjWithCaption."Object ID",AllObjWithCaption."Object Caption");
        END;
        // create actual table filter
        IF isCompiled AND NOT isEmpty AND isInLicense THEN
          ActualExportTableFilter += STRSUBSTNO('|%1',AllObjWithCaption."Object ID");
      UNTIL AllObjWithCaption.NEXT=0;
      ActualExportTableFilter := DELCHR(ActualExportTableFilter,'<','|');
    END;

    LOCAL PROCEDURE InitAvailableFields@55001(_ExportTableID@55006 : Integer);
    VAR
      ObjectNotInLicenseMsg@55002 : TextConst 'DEU=Achtung: Tabelle %1 - %2 ist nicht in der Lizenz.;ENU=Warning: Table %1 - %2 is not in license.';

      TableIsNotCompiledMsg@55004 : TextConst 'DEU=Achtung: Tabelle %1 ist nicht kompiliert;ENU=Warning: Table %1 is not compiled';
      field@55000 : Record 2000000041;
      _normalFields@55005 : TEMPORARY Record 2000000041;
      _blobAndMediaFields@55003 : TEMPORARY Record 2000000041;
    BEGIN
      SetObsoleteFieldFilter(field);
      SetTableFilterToExludeNonNormal(field);
      field.SETRANGE(TableNo,_ExportTableID);
      IF ExportFlowFields THEN
        field.SETFILTER(Class,'%1|%2',field.Class::Normal,field.Class::FlowField)
      ELSE
        field.SETRANGE(Class,field.Class::Normal);
      IF NOT ExportBlobContent THEN
        field.SETFILTER(Type,'<>26207&<>BLOB'); //26207=Media
      //* schema export - exclude meta info fields from field table
      IF _ExportTableID = DATABASE::Field THEN
          field.SETFILTER("No.",'<>%1&<>%2&<>%3&<>%4',24{OptionString},25{ObsoleteState},26{ObsoleteReason},27{DataClassification});
      IF field.FINDSET THEN REPEAT
        IF UPPERCASE(FORMAT(field.Type)) IN ['MEDIA','BLOB'] THEN BEGIN
          _blobAndMediaFields := field;
          _blobAndMediaFields.INSERT;
        END ELSE BEGIN
          _normalFields := field;
          _normalFields.INSERT;
        END;
      UNTIL field.NEXT=0;

      IF _ExportTableID = DATABASE::Field THEN
      AddCustomFieldsToSchemaExport(_normalFields);

      NormalFields.COPY(_normalFields,TRUE);
      BlobAndMediaFields.COPY(_blobAndMediaFields,TRUE);

      NoOfBlobAndMediaFields := BlobAndMediaFields.COUNT;
      NoOfNormalFields := NormalFields.COUNT;

      //* get last field no
      _normalFields.FINDLAST;
        LastFieldNoGlobal := _normalFields."No.";

      IF BlobAndMediaFields.FINDLAST THEN
        LastBlobFieldNoGlobal:= BlobAndMediaFields."No.";
    END;

    LOCAL PROCEDURE ReplaceStr@55017(_Text@1000000000 : Text;_From@1000000001 : Text;_To@1000000002 : Text) _NewText : Text;
    VAR
      _FromLength@55000 : Integer;
      _ToLength@55002 : Integer;
      _NextPos@55001 : Integer;
    BEGIN
      _FromLength := STRLEN(_From);
      _ToLength := STRLEN(_To);
      _NextPos := STRPOS(_Text,_From);
      WHILE _NextPos > 0 DO BEGIN
